feature -- Comparison

	frozen deep_equal (a: ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void
			-- or attached to isomorphic object structures?
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.is_deep_equal (b)
			end
		ensure -- from ANY
			shallow_implies_deep: standard_equal (a, b) implies Result
			both_or_none_void: (a = Void) implies (Result = (b = Void))
			same_type: (Result and (a /= Void)) implies (b /= Void and then a.same_type (b))
			symmetric: Result implies deep_equal (b, a)
		end

	frozen equal (a: ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void or attached
			-- to objects considered equal?
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.is_equal (b)
			end
		ensure -- from ANY
			definition: Result = (a = Void and b = Void) or else ((a /= Void and b /= Void) and then a.is_equal (b))
		end

	frozen is_deep_equal (other: DATABASE): BOOLEAN

	get_best_pointage
			-- Trouve le meilleur pointage dans `Current'
		local
			l_query: SQLITE_QUERY_STATEMENT
		do
			create l_query.make ("SELECT name, score FROM pointage ORDER BY score DESC;", db)
			l_query.execute (agent (ia_row: SQLITE_RESULT_ROW): BOOLEAN
				local
					j, j_count: NATURAL_32
				do
					from
						j := 1.to_natural_32
						j_count := ia_row.count
					until
						j > j_count
					loop
						print (ia_row.column_name (j))
						print (", ")
						print (ia_row.string_value (j))
						print ("%N")
						j := j + 1.to_natural_32
					end
					print ("%N")
				end)
		end

	insert_new_pointage (a_pointage: INTEGER_32; a_nom: STRING_8): INTEGER_32
			-- Id du joueur `a_nom' dans `Current'
		require
			a_pointage_is_at_least_0: a_pointage >= 0
			a_nom_is_not_empty: not a_nom.is_empty
		local
			l_id: INTEGER_32
			l_insert: SQLITE_INSERT_STATEMENT
			l_query: SQLITE_QUERY_STATEMENT
		do
			create l_insert.make ("INSERT INTO pointage (score,name) VALUES (?1, ?2);", db)
			check
				l_insert_is_compiled: l_insert.is_compiled
			end
			db.begin_transaction (False)
			l_insert.execute_with_arguments ([a_pointage.out, a_nom])
			db.commit
			create l_query.make ("SELECT `ID` from pointage where name='" + a_nom + "' ORDER BY id DESC LIMIT 1;", db)
			across
				l_query.execute_new as l_cursor
			loop
				l_id := l_cursor.item.integer_value (1.to_natural_32)
			end
			Result := l_id
		ensure
			result_is_at_least: Result >= 0
		end

	update_pointage (a_id: INTEGER_32; a_pointage: INTEGER_32; a_nom: STRING_8)
			-- Met à jour le pointage `a_pointage' associé au joueur `a_nom' dans la database `Current'
		require
			a_id_is_at_least_0: a_id >= 0
			a_pointage_is_at_least_0: a_pointage >= 0
			a_nom_is_not_empty: not a_nom.is_empty

			end

			l_modify.execute_with_arguments ([create {SQLITE_INTEGER_ARG}.make ("?1", a_pointage.to_integer_64), create {SQLITE_STRING_ARG}.make ("?2", a_nom), create {SQLITE_INTEGER_ARG}.make ("?3", a_id.to_integer_64)])

feature {NONE} -- Query

	frozen sqlite_exception (a_code: INTEGER_32; a_message: READABLE_STRING_8): SQLITE_EXCEPTION
			-- Retrieves an exception object based on the exception error code and message.
			--
			-- `a_code': A non-successful result code.
			-- `a_message': An optional exception message. NOt supplying an error message will generated a
			--              default error message.
			-- (from SQLITE_HELPERS)
		require -- from SQLITE_HELPERS
			not_a_code_sqlite_success: not sqlite_success (a_code)
			not_a_message_is_empty: attached a_message implies not a_message.is_empty
		do
			if (a_code & {SQLITE_RESULT_CODE}.e_io_err) = {SQLITE_RESULT_CODE}.e_io_err then
				if attached a_message then
					create {SQLITE_IO_EXCEPTION} Result.make_with_message (a_code, a_message)
				else
					create {SQLITE_IO_EXCEPTION} Result.make (a_code)
				end
			else
				if attached a_message then
					create Result.make_with_message (a_code, a_message)
				else
					create Result.make (a_code)
				end
			end
		end

	frozen sqlite_success (a_code: INTEGER_32): BOOLEAN
			-- Indicates if a result was a success or failure.
			--
			-- `a_code': A result code.
			-- `Result': True if the result code was a successful code; False for all errors codes.
			-- (from SQLITE_HELPERS)
		do
			Result := ({SQLITE_RESULT_CODE}.failure_mask & a_code) = 0 or else a_code = {SQLITE_RESULT_CODE}.row or a_code = {SQLITE_RESULT_CODE}.done
		ensure -- from SQLITE_HELPERS
			result_is_ok: Result implies (({SQLITE_RESULT_CODE}.failure_mask & a_code) = 0 or else a_code = {SQLITE_RESULT_CODE}.row or a_code = {SQLITE_RESULT_CODE}.done)
		end
