	description: "[Gestion de tous les bruits/sons/musiques du jeu]"
	author: "Tommy Teasdale"

	date: "15 Avril 2013"
	revision: "0.13.04.29"

class 
	BRUIT

create 
	make_sound,
	make_music

feature -- Access

	make_sound
		local
		do
		end

	make_music (a_type: STRING_8)
		local
			l_type, l_ctr: INTEGER_32
			l_default_format: NATURAL_16
		do
			if a_type = "ogg" then
				l_type := {SDL_MIXER}.mix_init_ogg
			elseif a_type = "flac" then
				l_type := {SDL_MIXER}.mix_init_flac
			elseif a_type = "mod" then
				l_type := {SDL_MIXER}.mix_init_mod
			else
				l_type := {SDL_MIXER}.mix_init_mp3
			end
			l_default_format := {SDL_MIXER}.mix_default_format
			l_ctr := {SDL_MIXER}.mix_init (l_type)
			l_ctr := {SDL_MIXER}.mix_openaudio (44100, l_default_format, 2, 1024)
		end

	music_load_file (a_file: STRING_8)
		local
			l_c_file: C_STRING
		do
			create l_c_file.make (a_file)
			music := {SDL_MIXER}.mix_loadmus (l_c_file.item)
		end

	music_play (a_loop: INTEGER_32)
			--if a_loop equal 0, loop forever
		local
			l_ctr: INTEGER_32
		do
			l_ctr := {SDL_MIXER}.mix_playmusic (music, a_loop)
		end

	music_pause
		do
			{SDL_MIXER}.mix_pausemusic
		end

	music_stop
		local
			l_ctr: INTEGER_32
		do
			l_ctr := {SDL_MIXER}.mix_haltmusic
		end

	music_set_volume (a_volume: INTEGER_32)
			--a_volume between 0 and 100
		local
			l_ctr: INTEGER_32
			l_real_volume: REAL_64
		do
			l_real_volume := (a_volume / 100) * 128.to_double
			volume := l_real_volume.truncated_to_integer
			l_ctr := {SDL_MIXER}.mix_volumemusic (volume)
		end

	music_get_volume: INTEGER_32
		local
			l_real_volume: REAL_64
		do
			l_real_volume := (volume / 128) * 100.to_double
			Result := l_real_volume.truncated_to_integer
		end

	music_close
		do
			music_stop
			{SDL_MIXER}.mix_closeaudio
		end

	music_quit
		do
			music_close
			{SDL_MIXER}.mix_quit
		end
	
feature {NONE} 

	music: POINTER

	sound: POINTER

	volume: INTEGER_32
	
end -- class BRUIT
