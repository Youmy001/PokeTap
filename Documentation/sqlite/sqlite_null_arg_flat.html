	frozen deep_equal (a: ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void
			-- or attached to isomorphic object structures?
			-- (from ANY)
		do
			if a = Void then
				Result := b = Void
			else
				Result := b /= Void and then a.is_deep_equal (b)
			end
		ensure -- from ANY
			shallow_implies_deep: standard_equal (a, b) implies Result
			both_or_none_void: (a = Void) implies (Result = (b = Void))
			same_type: (Result and (a /= Void)) implies (b /= Void and then a.same_type (b))
			symmetric: Result implies deep_equal (b, a)
		end

	frozen equal (a: ANY; b: like arg #1): BOOLEAN
			-- Are `a' and `b' either both void or attached
			-- to objects considered equal?
			-- (from ANY)
		do

				Result := b /= Void and then a.is_equal (b)
			end

		end

	is_valid_argument (a_value: ANY): BOOLEAN
			-- Indicates if the value is of an accepted type.
			--
			-- `a_value': A value to check.
			-- `Result': True if the value type is marshallable; False otherwise.
			-- (from SQLITE_BINDING_HELPERS)
		do
			Result := not attached a_value or else attached {READABLE_STRING_8} a_value or else attached {SQLITE_BIND_ARG [ANY]} a_value or else attached {MANAGED_POINTER} a_value
			if not Result and attached a_value then
				Result := True
				inspect Internal.dynamic_type (a_value)
				when {INTERNAL}.integer_8_type, {INTERNAL}.integer_16_type, {INTERNAL}.integer_32_type, {INTERNAL}.integer_64_type, {INTERNAL}.natural_8_type, {INTERNAL}.natural_16_type, {INTERNAL}.natural_32_type, {INTERNAL}.real_32_type, {INTERNAL}.real_64_type then 
					check
						Result
					end
				else
					Result := False
				end
			end
		end

	is_valid_arguments (a_args: TUPLE): BOOLEAN
			-- Determines if an argument tuple is valid.
			--
			-- `a_args': A tuple of arguments to validate.
			-- `Result': True if the supplied tuple contains valid value for arguments; False otherwise.
			-- (from SQLITE_BINDING_HELPERS)
		require -- from SQLITE_BINDING_HELPERS
			a_args_attached: attached a_args
			not_a_args_is_empty: not a_args.is_empty
		local
			i_count, i: INTEGER_32
		do
			Result := a_args.count <= sqlite_limit_variable_number
			from
				i := 1
				i_count := a_args.count
			until
				i > i_count or not Result
			loop
				Result := is_valid_argument (a_args [i])
				i := i + 1
			end
		ensure -- from SQLITE_BINDING_HELPERS
			a_args_small_enough: Result implies a_args.count <= sqlite_limit_variable_number
		end

	is_valid_variable_name (a_var: READABLE_STRING_8): BOOLEAN
			-- Determines if a SQLite variable name is valid.
			--
			-- `a_var': Name of the variable, including a qualifier.
			-- `Result': True if the variable is a valid SQLite variable; False otherwise
			-- (from SQLITE_BINDING_HELPERS)
		require -- from SQLITE_BINDING_HELPERS
			a_var_attached: attached a_var
			not_a_var_is_empty: not a_var.is_empty
		local
			l_qualifier, c: CHARACTER_8
			i_count, i: INTEGER_32
			l_index_str: READABLE_STRING_8
		do
			l_qualifier := a_var [1]
			if l_qualifier = '?' then
				i_count := a_var.count
				Result := i_count = 1
				if not Result then
					l_index_str := a_var.substring (2, i_count)
					if l_index_str.is_integer_32 then
						i := l_index_str.to_integer_32
						Result := i > 0 and i <= sqlite_limit_variable_number
					end
				end
			elseif l_qualifier = ':' or l_qualifier = '@' or l_qualifier = '$' then
				i_count := a_var.count
				Result := i_count > 1
				if Result then
					from
						i := 2
					until
						i > i_count or not Result
					loop
						c := a_var [i]
						Result := c.is_printable and not c.is_space
						i := i + 1
					end
				end
			end
		end

	frozen sqlite_exception (a_code: INTEGER_32; a_message: READABLE_STRING_8): SQLITE_EXCEPTION
			-- Retrieves an exception object based on the exception error code and message.
			--
			-- `a_code': A non-successful result code.
			-- `a_message': An optional exception message. NOt supplying an error message will generated a
			--              default error message.
			-- (from SQLITE_HELPERS)
		require -- from SQLITE_HELPERS
			not_a_code_sqlite_success: not sqlite_success (a_code)
			not_a_message_is_empty: attached a_message implies not a_message.is_empty
		do
			if (a_code & {SQLITE_RESULT_CODE}.e_io_err) = {SQLITE_RESULT_CODE}.e_io_err then
				if attached a_message then
					create {SQLITE_IO_EXCEPTION} Result.make_with_message (a_code, a_message)
				else
					create {SQLITE_IO_EXCEPTION} Result.make (a_code)
				end
			else
				if attached a_message then
					create Result.make_with_message (a_code, a_message)
				else
					create Result.make (a_code)
				end
			end
		end

	frozen sqlite_success (a_code: INTEGER_32): BOOLEAN
			-- Indicates if a result was a success or failure.
			--
			-- `a_code': A result code.
			-- `Result': True if the result code was a successful code; False for all errors codes.
			-- (from SQLITE_HELPERS)
		do
			Result := ({SQLITE_RESULT_CODE}.failure_mask & a_code) = 0 or else a_code = {SQLITE_RESULT_CODE}.row or a_code = {SQLITE_RESULT_CODE}.done
		ensure -- from SQLITE_HELPERS
			result_is_ok: Result implies (({SQLITE_RESULT_CODE}.failure_mask & a_code) = 0 or else a_code = {SQLITE_RESULT_CODE}.row or a_code = {SQLITE_RESULT_CODE}.done)
		end
