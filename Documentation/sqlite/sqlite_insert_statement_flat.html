	is_valid_arguments (a_args: TUPLE): BOOLEAN
			-- Determines if an argument tuple is valid.
			--
			-- `a_args': A tuple of arguments to validate.
			-- `Result': True if the supplied tuple contains valid value for arguments; False otherwise.
			-- (from SQLITE_BINDING_HELPERS)
		require -- from SQLITE_BINDING_HELPERS
			a_args_attached: attached a_args
			not_a_args_is_empty: not a_args.is_empty
		local
			i_count, i: INTEGER_32
		do
			Result := a_args.count <= sqlite_limit_variable_number
			from
				i := 1
				i_count := a_args.count
			until
				i > i_count or not Result
			loop
				Result := is_valid_argument (a_args [i])

			end

		end

	is_valid_variable_name (a_var: READABLE_STRING_8): BOOLEAN
			-- Determines if a SQLite variable name is valid.
			--
			-- `a_var': Name of the variable, including a qualifier.
			-- `Result': True if the variable is a valid SQLite variable; False otherwise
			-- (from SQLITE_BINDING_HELPERS)
		require -- from SQLITE_BINDING_HELPERS
			a_var_attached: attached a_var
			not_a_var_is_empty: not a_var.is_empty
		local
			l_qualifier, c: CHARACTER_8
			i_count, i: INTEGER_32
			l_index_str: READABLE_STRING_8
		do
			l_qualifier := a_var [1]
			if l_qualifier = '?' then
				i_count := a_var.count
				Result := i_count = 1
				if not Result then
					l_index_str := a_var.substring (2, i_count)
					if l_index_str.is_integer_32 then
						i := l_index_str.to_integer_32
						Result := i > 0 and i <= sqlite_limit_variable_number
					end
				end
			elseif l_qualifier = ':' or l_qualifier = '@' or l_qualifier = '$' then
				i_count := a_var.count
				Result := i_count > 1
				if Result then
					from
						i := 2
					until
						i > i_count or not Result
					loop
						c := a_var [i]
						Result := c.is_printable and not c.is_space
						i := i + 1
					end
				end
			end
		end

feature {NONE} -- Basic operations: Compilation

	compile
			-- Compiles the current statement.
			-- (from SQLITE_STATEMENT)
		require -- from SQLITE_STATEMENT
			is_sqlite_available: Is_sqlite_available
			is_interface_usable: is_interface_usable
			not_is_connected: not is_connected
			not_has_error: not has_error
			database_is_readable: database.is_readable
			database: database.is_readable
		local
			l_db: like database
			l_string: C_STRING
			l_next_string: STRING_8
			l_next_statement: SQLITE_STATEMENT
			l_stmt_handle: POINTER
			l_internal_db: POINTER
			l_tail: POINTER
			l_locked: BOOLEAN
			l_result: INTEGER_32
		do
			create l_string.make (compile_statement_string)
			l_db := database
			l_db.lock
			l_locked := True
			l_internal_db := l_db.internal_db
			l_result := sqlite3_prepare_v2 (Sqlite_api, l_internal_db, l_string.item, l_string.count + 1, $l_stmt_handle, $l_tail)
			if l_stmt_handle /= default_pointer then
				internal_stmt := l_stmt_handle
				internal_db := l_internal_db
				last_exception := Void
				if l_tail /= default_pointer then
					check
						next_statement_detached: not attached next_statement
					end
					create l_string.make_shared_from_pointer (l_tail)
					l_next_string := l_string.string
					if not l_next_string.is_empty then
						create statement_string.make_from_string (statement_string.substring (1, statement_string.count - l_next_string.count))
						l_next_string.prune_all_leading (';')
						l_next_string.left_adjust
						if not l_next_string.is_empty then
							create l_next_statement.make (l_next_string, l_db)
							if l_next_statement.has_error then
								reset_compilation_data
								last_exception := l_next_statement.last_exception
							else
								next_statement := l_next_statement
							end
						end
					end
				end
			else
				reset_compilation_data
				if sqlite_success (l_result) then
					last_exception := sqlite_exception ({SQLITE_RESULT_CODE}.e_misuse, "no statement: syntax error")
				else
					last_exception := l_db.last_exception
				end
			end
			l_locked := False
			l_db.unlock
		ensure -- from SQLITE_STATEMENT
			not_internal_db_is_null: is_compiled implies internal_db /= default_pointer
			internal_db_is_null: not is_compiled implies internal_db = default_pointer
			last_exception_attached: not is_compiled implies attached last_exception
			last_exception_attached: is_compiled implies not attached last_exception
		rescue
			if l_locked then
				l_locked := False
				database.unlock
			end
		end

	reset_compilation_data
			-- Clears any cached information related to the preperation/compilation of the statement
			-- (from SQLITE_STATEMENT)
		do
			internal_stmt := default_pointer
			internal_db := default_pointer
			reset
			mark := mark + 1.to_natural_32
		ensure -- from SQLITE_STATEMENT
			not_is_compiled: not is_compiled
			not_is_connected: not is_connected
			not_is_abort_requested: not is_abort_requested
			internal_changes_count_reset: internal_changes_count = 0.to_natural_32
			last_exception_detached: not attached last_exception
			mark_increased: mark > old mark
			not_has_error: not has_error
		end
	
feature {SQLITE_STATEMENT} -- Basic operations: Execution

	execute_internal (a_callback: FUNCTION [ANY, TUPLE [row: SQLITE_RESULT_ROW], BOOLEAN]; a_bindings: ARRAY [SQLITE_BIND_ARG [ANY]])
			-- Performs execution of the SQLite statement with a callback routine for each returned result row.
			--
			-- `a_callback': A callback routine accepting a result row as its argument.
			--               Return True from the function to abort further calls when there is more result data.
			-- `a_bindings': An array or binding arguments, or Void if the statement does not require any arguments.
			-- (from SQLITE_STATEMENT)
		require -- from SQLITE_STATEMENT
			is_sqlite_available: Is_sqlite_available
			is_interface_usable: is_interface_usable
			is_compiled: is_compiled
			is_connected: is_connected
			not_is_executing: not is_executing
			is_accessible: is_accessible
			database_is_readable: database.is_readable
			a_bindings_attached: has_arguments implies attached a_bindings
			a_bindings_count_big_enough: not attached a_bindings or else a_bindings.count.as_natural_32 = arguments_count
		local
			l_api: like Sqlite_api
			l_stmt: like internal_stmt
			l_db: like database
			l_exception: SQLITE_EXCEPTION
			l_result: INTEGER_32
			l_done: BOOLEAN
			l_locked: BOOLEAN
			l_row: SQLITE_RESULT_ROW
			i_upper, i: INTEGER_32
			l_arg_variable: IMMUTABLE_STRING_8
			l_arg_id: C_STRING
			l_arg_index: INTEGER_32
			l_total_count: NATURAL_32
		do
			if attached a_bindings as l_bindings then
				from
					i := l_bindings.lower
					i_upper := l_bindings.upper
				until
					i > i_upper
				loop
					if attached l_bindings [i] as l_arg then
						l_arg_variable := l_arg.variable
						create l_arg_id.make (l_arg_variable)
						l_arg_index := sqlite3_bind_parameter_index (Sqlite_api, internal_stmt, l_arg_id.item)
						if l_arg_index = 0 and then l_arg_variable [1] = '?' then
							l_arg_variable := l_arg_variable.substring (2, l_arg_variable.count)
							if l_arg_variable.is_integer_32 then
								l_arg_index := l_arg_variable.to_integer_32
							else
								check
									should_never_happen: False
								end
							end
						end
						if l_arg_index > 0 then
							l_arg.bind_to_statement (Current, l_arg_index)
						end
					end
					i := i + 1
				end
			end
			mark := mark + 1.to_natural_32
			l_api := Sqlite_api
			l_stmt := internal_stmt
			l_db := database
			reset_all
			is_executing := True
			on_before_execute
			l_db.lock
			l_locked := True
			l_total_count := l_db.total_changes_count
			from
				create l_row.make (Current)
				l_result := sqlite3_step (l_api, l_stmt)
				l_done := l_result = {SQLITE_RESULT_CODE}.done
			until
				not sqlite_success (l_result) or l_done
			loop
				l_db.unlock
				l_locked := False
				if attached a_callback then
					l_row.index := l_row.index + 1.to_natural_32
					is_abort_requested := a_callback.item ([l_row]) or is_abort_requested
				end
				if is_connected then
					l_db.lock
					l_locked := True
					l_done := is_abort_requested
					if l_done then
						l_db.abort
					else
						l_done := l_result /= {SQLITE_RESULT_CODE}.row
						if not l_done then
							l_result := sqlite3_step (l_api, l_stmt)
							l_done := l_result = {SQLITE_RESULT_CODE}.done
						end
					end
				else
					l_done := True
				end
			end
			if not sqlite_success (l_result) then
				if is_connected then
					l_exception := l_db.last_exception
					if l_locked then
						l_locked := False
						l_db.unlock
					end
				end
				if not attached l_exception then
					l_exception := sqlite_exception (l_result, Void)
				end
				last_exception := l_exception
				on_after_executed
				l_exception.raise
			else
				internal_changes_count := l_db.total_changes_count - l_total_count
				if l_locked then
					l_locked := False
					l_db.unlock
				end
				if attached next_statement as l_next then
					l_next.execute_internal (a_callback, a_bindings)
				end
				on_after_executed
			end
			is_executing := False
			l_result := sqlite3_reset (l_api, l_stmt)
			check
				success: sqlite_success (l_result)
			end
			if attached a_bindings then
				l_result := sqlite3_clear_bindings (l_api, l_stmt)
				check
					success: sqlite_success (l_result)
				end
			end
		ensure -- from SQLITE_STATEMENT
			not_is_executing: not is_executing
			mark_increased: mark > old mark
		rescue
			if l_locked then
				check
					l_db_attached: attached l_db
				end
				l_locked := False
				l_db.unlock
			end
			if is_executing then
				is_executing := False
				sqlite3_reset (Sqlite_api, internal_stmt).do_nothing
			end
		end

	reset
			-- Resets any cached information from the last execution.
			-- (from SQLITE_STATEMENT)
		require -- from SQLITE_STATEMENT
			not_is_executing: not is_executing
		do
			is_abort_requested := False
			internal_changes_count := 0.to_natural_32
			last_exception := Void
		ensure -- from SQLITE_STATEMENT
			not_is_abort_requested: not is_abort_requested
			internal_changes_count_reset: internal_changes_count = 0.to_natural_32
			last_exception_detached: not attached last_exception
			not_has_error: not has_error
		end

	reset_all
			-- Resets any cached information from the last execution for all statements.
			-- (from SQLITE_STATEMENT)
		require -- from SQLITE_STATEMENT
			not_is_executing: not is_executing
		do
			reset
			if attached next_statement as l_next then
				l_next.reset_all
			end
		ensure -- from SQLITE_STATEMENT
			not_is_abort_requested: not is_abort_requested
			internal_changes_count_reset: internal_changes_count = 0.to_natural_32
			last_exception_detached: not attached last_exception
			not_has_error: not has_error
		end

	dispose
			-- Action to be executed just before garbage collection
			-- reclaims an object.
			-- Effect it in descendants to perform specific dispose
			-- actions. Those actions should only take care of freeing
			-- external resources; they should not perform remote calls
			-- on other objects since these may also be dead and reclaimed.
			-- (from SQLITE_STATEMENT)
		require -- from  DISPOSABLE
			True
		local
			l_stmt: like internal_stmt
			l_db: like internal_db
			l_api: like Sqlite_api
			l_result: INTEGER_32
		do
			if not is_in_final_collect then
				l_stmt := internal_stmt
				if l_stmt /= default_pointer and then Is_sqlite_available then
					l_api := Sqlite_api
					l_db := sqlite3_db_handle (l_api, l_stmt)
					if l_db /= default_pointer then
						if l_db = internal_db then
							l_result := sqlite3_finalize (l_api, l_stmt)
							if (l_result & {SQLITE_RESULT_CODE}.mask) = {SQLITE_RESULT_CODE}.e_misuse then
								check
									is_closed: database.is_closed
								end
							else
								check
									succeess: sqlite_success (l_result)
								end
							end
						else
							check
								False
							end
						end
					end
				end
				internal_stmt := default_pointer
				internal_db := default_pointer
			end
		ensure then -- from SQLITE_STATEMENT
			not_is_compiled: not is_compiled
			not_is_connected: not is_connected
			internal_stmt_is_null: internal_stmt = default_pointer
			internal_db_is_null: internal_db = default_pointer
		end

	new_binding_argument (a_value: ANY; a_var: READABLE_STRING_8): SQLITE_BIND_ARG [ANY]
			-- Creates a new SQLite bind argument for us with executing SQLite statements containing arguments.
			--
			-- `a_value': A value to create a binding argument for.
			-- `Result': A bound argument, or `a_value' if the argument is already a binding argument.
			-- (from SQLITE_BIND_ARG_MARSHALLER)
		require -- from SQLITE_BIND_ARG_MARSHALLER
			a_value_is_valid_argument: is_valid_argument (a_value)
			a_var_attached: attached a_var
			not_a_var_is_empty: not a_var.is_empty
			a_var_is_valid_variable_name: is_valid_variable_name (a_var)
		local
			l_scalar_result: SQLITE_BIND_ARG [ANY]
			l_type_id: INTEGER_32
		do
			if attached a_value then
				if attached {SQLITE_BIND_ARG [ANY]} a_value as l_result then
					Result := l_result
				elseif attached {READABLE_STRING_8} a_value as l_value then
					create {SQLITE_STRING_ARG} Result.make (a_var, l_value)
				elseif attached {MANAGED_POINTER} a_value as l_value then
					create {SQLITE_BLOB_ARG} Result.make (a_var, l_value)
				else
					l_type_id := Internal.dynamic_type (a_value)
					if l_type_id = ({INTEGER_8}).type_id then
						if attached {INTEGER_8_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.to_integer_64)
						end
					elseif l_type_id = ({INTEGER_16}).type_id then
						if attached {INTEGER_16_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.to_integer_64)
						end
					elseif l_type_id = ({INTEGER_32}).type_id then
						if attached {INTEGER_32_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.to_integer_64)
						end
					elseif l_type_id = ({INTEGER_64}).type_id then
						if attached {INTEGER_64_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item)
						end
					elseif l_type_id = ({NATURAL_8}).type_id then
						if attached {NATURAL_8_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.as_integer_32.to_integer_64)
						end
					elseif l_type_id = ({NATURAL_16}).type_id then
						if attached {NATURAL_16_REF} a_value as l_value then
							create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.as_integer_32.to_integer_64)
						end
					elseif l_type_id = ({NATURAL_32}).type_id then
						if attached {NATURAL_32_REF} a_value as l_value then
							if l_value <= {INTEGER_32}.max_value.as_natural_32 then
								create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.as_integer_32.to_integer_64)
							else
								create {SQLITE_INTEGER_ARG} l_scalar_result.make (a_var, l_value.item.as_integer_64)
							end
						end
					elseif l_type_id = ({REAL_32}).type_id then
						if attached {REAL_32_REF} a_value as l_value then
							create {SQLITE_DOUBLE_ARG} l_scalar_result.make (a_var, l_value.item.to_double)
						end
					elseif l_type_id = ({REAL_64}).type_id then
						if attached {REAL_64_REF} a_value as l_value then
							create {SQLITE_DOUBLE_ARG} l_scalar_result.make (a_var, l_value.item)
						end
					else
						check
							unsupported_type: False
						end
					end
					check
						l_scalar_result_attached: attached l_scalar_result
					end
					Result := l_scalar_result
				end
			else
				create {SQLITE_NULL_ARG} Result.make (a_var)
			end
		ensure -- from SQLITE_BIND_ARG_MARSHALLER
			result_attached: attached Result
			result_is_value_marshalled: is_value_marshalled (Result)
		end

	new_binding_argument_array (a_args: TUPLE): ARRAY [SQLITE_BIND_ARG [ANY]]
			-- Creates a new argument array from a tuple of values.
			--
			-- `a_args': Arguments to create a new array for.
			-- `Result': Argument array to use with processing
			-- (from SQLITE_BIND_ARG_MARSHALLER)
		require -- from SQLITE_BIND_ARG_MARSHALLER
			a_args_attached: attached a_args
			not_a_args_is_empty: not a_args.is_empty
			a_args_is_valid_arguments: is_valid_arguments (a_args)
		local
			i_count, i: INTEGER_32
			l_arg: ANY
			l_var: STRING_8
		do
			i_count := a_args.count.min (sqlite_limit_variable_number)
			create Result.make_filled (Default_bind_arg, 1, i_count)
			from
				i := 1
			until
				i > i_count
			loop
				l_arg := a_args [i]
				if attached {SQLITE_BIND_ARG [ANY]} l_arg as l_bind_arg then
					Result [i] := l_bind_arg
				else
					create l_var.make (3)
					l_var.append_character ('?')
					l_var.append_integer (i)
					Result [i] := new_binding_argument (l_arg, l_var)
				end
				i := i + 1
			end
		ensure -- from SQLITE_BIND_ARG_MARSHALLER
			result_attached: attached Result
			result_count_matches: Result.count = a_args.count
		end
