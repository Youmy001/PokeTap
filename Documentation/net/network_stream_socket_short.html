note
	description: "A network stream socket."
	legal: "See notice at end of class."
	status: "See notice at end of class."
	date: "$Date: 2011-10-24 16:52:05 -0700 (Mon, 24 Oct 2011) $"
	revision: "$Revision: 87603 $"

class interface
	NETWORK_STREAM_SOCKET

create 
	make,
	make_client_by_port,
	make_client_by_address_and_port,
	make_server_by_port,
	make_loopback_server_by_port


create {NETWORK_STREAM_SOCKET}
	make_from_descriptor_and_address,
	create_from_descriptor

feature -- Initialization

	make
			-- Create a network stream socket.
		ensure
			timeout_set_to_default: timeout = Default_timeout

	make_client_by_port (a_peer_port: INTEGER_32; a_peer_host: STRING_8)
			-- Create a client connection to `a_peer_host' on
			-- `a_peer_port'.
		require
			valid_peer_host: a_peer_host /= Void and then not a_peer_host.is_empty
			valid_port: a_peer_port >= 0

	make_server_by_port (a_port: INTEGER_32)
			-- Create server socket on `a_port' and on any incoming address.
		require
			valid_port: a_port >= 0

	make_loopback_server_by_port (a_port: INTEGER_32)
			-- Create server socket on `a_port' that listen only for loopback interface peer.
		require
			valid_port: a_port >= 0

	make_client_by_address_and_port (a_peer_address: INET_ADDRESS; a_peer_port: INTEGER_32)
			-- Create a client connection to `a_peer_host' on
			-- `a_peer_port'.
		require
			valid_peer_address: a_peer_address /= Void
			valid_port: a_peer_port >= 0
	
feature 

	connect_timeout: INTEGER_32
			-- The connect timeout in milliseconds

	set_connect_timeout (a_timeout: INTEGER_32)
			--  Sets the connect timeout in milliseconds

	accept_timeout: INTEGER_32
			-- The connect timeout in milliseconds

	set_accept_timeout (a_timeout: INTEGER_32)
			--  Sets the accept timeout in milliseconds

	listen (queue: INTEGER_32)
			-- Listen on socket for at most `queue' connections.

	accept
			-- Accept a new connection on listen socket.
			-- Accepted service socket available in accepted.
	
feature -- Status report

	maximum_seg_size: INTEGER_32
			-- Maximum segment size
		require
			socket_exists: exists

	has_delay: BOOLEAN
			-- Is option TCPNO_DELAY off.
		require
			socket_exists: exists

	is_linger_on: BOOLEAN
			-- Is lingering switched on?
		require
			socket_exists: exists

	is_out_of_band_inline: BOOLEAN
			-- Are out-of-band packets sent inline?
		require
			socket_exists: exists

	linger_time: INTEGER_32
			-- Linger time
		require
			socket_exists: exists
	
feature -- Status setting

	set_delay
			-- Switch option TCPNO_DELAY off.
		require
			socket_exists: exists

	set_nodelay
			-- Switch option TCPNO_DELAY on.
		require
			socket_exists: exists

	set_linger_on (time: INTEGER_32)
			-- Switch lingering on and set linger time to `time'.
		require
			socket_exists: exists

	set_linger_off
			-- Switch lingering off.
		require
			socket_exists: exists

	set_out_of_band_inline
			-- Switch "out of band packets inline" on.
		require
			socket_exists: exists

	set_out_of_band_not_inline
			-- Switch "out of band packets inline" off.
		require
			socket_exists: exists
